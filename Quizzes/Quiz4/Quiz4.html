currencies["bool_cond"] = (currencies["growth_euro"] > 2) | (currencies["growth_euro"] < -2)


plt.plot("date", "growth_euro", data = currencies, color = "pink")
plt.xlabel("Time")
plt.ylabel("Daily percentage change in exchange rate of the Euro")
plt.title("Exchange rate of the Euro over time")

vec_y = currencies["growth_euro"]
plt.fill_between(x= currencies["date"],
                 y1 = vec_y.min(),
                 y2 = vec_y.max(),
                 where = currencies["bool_cond"],
                 alpha = 0.2,color = "blue")

plt.show()

month_config = pd.Grouper(key='date', freq='m')

monthlydata = (currencies
               .groupby(month_config)
               .agg(eur_to_usd_stdev = ("eur_to_usd","std"))
               .reset_index())
